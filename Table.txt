CREATE TABLE students (
    name VARCHAR2(100) NOT NULL,
    dob DATE NOT NULL,
    gender VARCHAR2(10) NOT NULL,
    email VARCHAR2(100) UNIQUE NOT NULL,
    phone VARCHAR2(20) NOT NULL,
    admission_number VARCHAR2(50) UNIQUE NOT NULL,
    course VARCHAR2(100) NOT NULL,
    address VARCHAR2(255) NOT NULL,
    password VARCHAR2(100) NOT NULL
);

CREATE TABLE staff (
    name VARCHAR2(255) NOT NULL,
    email VARCHAR2(255) NOT NULL UNIQUE,
    password VARCHAR2(255) NOT NULL,
    dob DATE,
    gender VARCHAR2(50),
    phone VARCHAR2(50),
    address VARCHAR2(255),
    position VARCHAR2(50),
    branch VARCHAR2(50)
);


CREATE TABLE cookie (
    email VARCHAR(255) NOT NULL,
    session_id VARCHAR(255) NOT NULL);


CREATE TABLE scheduled_resources (
    id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
    resource_type VARCHAR2(50) NOT NULL,
    resource_details VARCHAR2(100) NOT NULL,
    booked_date DATE NOT NULL,
    from_time TIMESTAMP NOT NULL,
    to_time TIMESTAMP NOT NULL,
    purpose CLOB NOT NULL,
    user_type VARCHAR2(50) NOT NULL,
    email VARCHAR2(100) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


CREATE TABLE registrations (
    email_id VARCHAR2(255) NOT NULL,
    event_id NUMBER NOT NULL
    );


CREATE TABLE tasks (
    id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name VARCHAR2(255) NOT NULL,
    description VARCHAR2(255) NOT NULL,
    deadline DATE NOT NULL,
    assigned_to VARCHAR2(255) NOT NULL,
    assigned_by VARCHAR2(255) NOT NULL
);


CREATE TABLE communications (
    id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    title VARCHAR2(255) NOT NULL,
    message CLOB NOT NULL,
    sender VARCHAR2(255) NOT NULL,
    recipient VARCHAR2(255) NOT NULL,
    timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
